async function fetchData() { try { console.log("A procurar dados econômicos..."); // URLs das APIs const urlSelic = 'https://api.bcb.gov.br'; const urlIpca = 'https://api.bcb.gov.br'; const urlDolar = 'https://economia.awesomeapi.com.br'; const [resSelic, resIpca, resDolar] = await Promise.all([ fetch(urlSelic).catch(() => ({ json: () => [{ valor: "13.25" }] })), // Plano B se falhar fetch(urlIpca).catch(() => ({ json: () => Array(12).fill({ valor: "0.4" }) })), fetch(urlDolar).catch(() => ({ json: () => ({ USDBRL: { bid: "5.00" } }) })) ]); const selicData = await resSelic.json(); const ipcaData = await resIpca.json(); const dolarData = await resDolar.json(); // 1. Processar Selic const selic = selicData[0] ? parseFloat(selicData[0].valor).toFixed(2) : "13.25"; // 2. Processar IPCA (Soma simples dos últimos 12 meses para estimativa) const ipcaAcumulado = ipcaData.length > 0 ? ipcaData.reduce((acc, curr) => acc + parseFloat(curr.valor), 0).toFixed(2) : "4.50"; // 3. Processar Dólar const dolarRaw = dolarData.USDBRL ? dolarData.USDBRL.bid : "5.00"; const dolarFormatado = parseFloat(dolarRaw).toLocaleString('pt-BR', { style: 'currency', currency: 'BRL' }); // ATUALIZAR INTERFACE document.getElementById('txt-selic').innerText = selic + '%'; document.getElementById('tab-selic').innerText = selic + '%'; document.getElementById('live-selic').innerText = selic + '%'; document.getElementById('bar-selic').style.width = Math.min(selic * 7, 100) + '%'; document.getElementById('txt-ipca').innerText = ipcaAcumulado + '%'; document.getElementById('tab-ipca').innerText = ipcaAcumulado + '%'; document.getElementById('bar-ipca').style.width = Math.min(ipcaAcumulado * 10, 100) + '%'; document.getElementById('live-dolar').innerText = dolarFormatado; document.getElementById('bar-trabalho').style.width = '92%'; console.log("Dados atualizados com sucesso!"); } catch (e) { console.error("Erro crítico ao carregar dados:", e); // Fallback visual para o usuário não ver "--" document.getElementById('live-selic').innerText = "Erro API"; } }
